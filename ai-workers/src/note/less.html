全局配置less文件

在 vite.config.js 文件中配置css 项，其中文件路径指向全局less文件即可全局使用变量 

            import { defineConfig } from 'vite';
            import vue from '@vitejs/plugin-vue';
            export default defineConfig({
                plugins: [
                    vue(),
                ],
                // less 配置
                css: {
                    preprocessorOptions: {
                        less: {
                            // less文件路径
                            additionalData: `@import "src/assets/css/theme.less";`,
                            // 这个设置的意思是告诉构建工具在处理Less文件时启用JavaScript功能。在Less文件中，有时候可能需要使用JavaScript代码，比如函数调用或者变量定义等，通过设置javascriptEnabled为true，可以让Less文件支持这些特性。
                            javascriptEnabled: true,
                        }
                    }
                },
            })
