全局配置less文件

在 vite.config.js 文件中配置css 项，其中文件路径指向全局less文件即可全局使用变量 

            import { defineConfig } from 'vite';
            import vue from '@vitejs/plugin-vue';
            export default defineConfig({
                plugins: [
                    vue(),
                ],
                // less 配置
                css: {
                    preprocessorOptions: {
                        less: {
                            // less文件路径
                            additionalData: `@import "src/assets/css/theme.less";`,
                        }
                    }
                },
            })
